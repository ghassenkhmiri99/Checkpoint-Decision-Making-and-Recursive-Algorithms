function fibonacci(n) {
    let fibArray = [0, 1];

    for (let i = 2; i < n + 1; i++) {
        fibArray[i] = fibArray[i - 1] + fibArray[i - 2];
    }

    let result = new Array(n + 1);
    for (let i = 0; i < n + 1; i++) {
        result[i] = fibArray[i];
    }

    console.log(result);
}

palindrome method :
function isPalindrome(string) {
    let rev = "";
    for (let i = string.length - 1; i >= 0; i--) {
        rev += string[i];
    }
    if (rev == string) {
        return true
    } else {
        return false;
    }
}
console.log(isPalindrome(prompt()));
